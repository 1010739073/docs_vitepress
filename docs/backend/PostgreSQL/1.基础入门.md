# PostgreSQL 基础入门

## 一、PostgreSQL 安装与配置

### 1.1 macOS 安装

```bash
# Homebrew 安装（推荐）
brew install postgresql@16

# 启动服务
brew services start postgresql@16

# 验证安装
psql --version
```

### 1.2 Linux 安装（Ubuntu/Debian）

```bash
# 添加官方仓库
sudo sh -c 'echo "deb https://apt.postgresql.org/pub/repos/apt $(lsb_release -cs)-pgdg main" > /etc/apt/sources.list.d/pgdg.list'
wget --quiet -O - https://www.postgresql.org/media/keys/ACCC4CF8.asc | sudo apt-key add -

# 安装
sudo apt-get update
sudo apt-get install -y postgresql-16

# 启动服务
sudo systemctl start postgresql
sudo systemctl enable postgresql
```

### 1.3 Docker 安装

```bash
docker pull postgres:16

docker run -d \
  --name postgres \
  -e POSTGRES_USER=myuser \
  -e POSTGRES_PASSWORD=mypassword \
  -e POSTGRES_DB=mydb \
  -p 5432:5432 \
  -v pgdata:/var/lib/postgresql/data \
  postgres:16
```

### 1.4 初始配置

```bash
# 切换到 postgres 用户
sudo -u postgres psql

# 或直接连接
psql -h localhost -U myuser -d mydb
```

## 二、数据库和表的创建

### 2.1 数据库管理

```sql
-- 创建数据库
CREATE DATABASE mydb
    ENCODING 'UTF8'
    LC_COLLATE 'zh_CN.UTF-8'
    LC_CTYPE 'zh_CN.UTF-8'
    TEMPLATE template0;

-- 删除数据库（需断开所有连接）
DROP DATABASE IF EXISTS mydb;

-- 重命名数据库
ALTER DATABASE mydb RENAME TO newdb;

-- 查看当前数据库
SELECT current_database();
```

### 2.2 创建表

```sql
-- 创建用户表
CREATE TABLE users (
    id SERIAL PRIMARY KEY,
    username VARCHAR(50) NOT NULL UNIQUE,
    email VARCHAR(100) NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    age INT CHECK (age > 0 AND age < 150),
    is_active BOOLEAN DEFAULT TRUE,
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 创建订单表（含外键）
CREATE TABLE orders (
    id SERIAL PRIMARY KEY,
    user_id INT NOT NULL REFERENCES users(id) ON DELETE CASCADE,
    order_number VARCHAR(50) NOT NULL UNIQUE,
    total_amount NUMERIC(10,2) NOT NULL DEFAULT 0,
    status VARCHAR(20) DEFAULT 'pending',
    created_at TIMESTAMP DEFAULT NOW()
);

-- 查看表结构
\d users
\d+ orders
```

### 2.3 Schema 管理

```sql
-- PostgreSQL 中 Schema 相当于命名空间
CREATE SCHEMA business;

-- 在不同 Schema 下创建表
CREATE TABLE business.customers (
    id SERIAL PRIMARY KEY,
    name VARCHAR(100) NOT NULL
);

-- 设置搜索路径
SET search_path TO business, public;
SHOW search_path;

-- 查看所有 Schema
\dn
```

## 三、基本 CRUD 操作

### 3.1 插入数据

```sql
-- 单行插入
INSERT INTO users (username, email, password_hash, age)
VALUES ('zhangsan', 'zhangsan@example.com', 'hashed_pw_1', 28);

-- 多行插入
INSERT INTO users (username, email, password_hash, age) VALUES
    ('lisi', 'lisi@example.com', 'hashed_pw_2', 32),
    ('wangwu', 'wangwu@example.com', 'hashed_pw_3', 25);

-- 插入并返回生成的值
INSERT INTO users (username, email, password_hash, age)
VALUES ('zhaoliu', 'zhaoliu@example.com', 'hashed_pw_4', 30)
RETURNING id, created_at;

-- INSERT ... ON CONFLICT (UPSERT)
INSERT INTO users (username, email, password_hash, age)
VALUES ('zhangsan', 'new_email@example.com', 'new_hash', 29)
ON CONFLICT (username)
DO UPDATE SET email = EXCLUDED.email, age = EXCLUDED.age;
```

### 3.2 查询数据

```sql
-- 基本查询
SELECT id, username, email FROM users;

-- 条件查询
SELECT * FROM users WHERE age > 25 AND is_active = TRUE;

-- 模糊查询
SELECT * FROM users WHERE email LIKE '%@example.com';

-- 排序
SELECT * FROM users ORDER BY created_at DESC LIMIT 10;

-- 分页
SELECT * FROM users ORDER BY id LIMIT 10 OFFSET 20;

-- 聚合查询
SELECT is_active, COUNT(*), AVG(age) FROM users GROUP BY is_active;

-- 连接查询
SELECT u.username, o.order_number, o.total_amount
FROM users u
INNER JOIN orders o ON u.id = o.user_id
WHERE o.status = 'pending';
```

### 3.3 更新数据

```sql
-- 基本更新
UPDATE users SET age = 31, updated_at = NOW() WHERE id = 1;

-- 多表更新
UPDATE orders o SET total_amount = total_amount * 1.1
FROM users u
WHERE o.user_id = u.id AND u.is_active = TRUE;
```

### 3.4 删除数据

```sql
-- 条件删除
DELETE FROM users WHERE id = 1;

-- 清空表（比 DELETE 快，不触发触发器）
TRUNCATE TABLE orders;

-- 清空表并重置序列
TRUNCATE TABLE orders RESTART IDENTITY CASCADE;

-- 软删除（推荐）
UPDATE users SET is_active = FALSE WHERE id = 1;
```

## 四、数据类型详解

| 类型 | 存储大小 | 说明 |
|-----|---------|------|
| **SMALLINT** | 2 字节 | 小整数，-32768 到 32767 |
| **INT / INTEGER** | 4 字节 | 整数，-21 亿到 21 亿 |
| **BIGINT** | 8 字节 | 大整数 |
| **SERIAL** | 4 字节 | 自增整数 |
| **BIGSERIAL** | 8 字节 | 自增大整数 |
| **NUMERIC(p,s)** | 可变 | 精确小数，p 总位数，s 小数位 |
| **REAL / DOUBLE** | 4/8 字节 | 浮点数 |
| **CHAR(n)** | 定长 | 定长字符串（不足补空格） |
| **VARCHAR(n)** | 可变 | 变长字符串 |
| **TEXT** | 可变 | 无限制文本 |
| **BOOLEAN** | 1 字节 | true / false / null |
| **DATE** | 4 字节 | 日期 |
| **TIME** | 8 字节 | 时间 |
| **TIMESTAMP** | 8 字节 | 日期时间（无时区） |
| **TIMESTAMPTZ** | 8 字节 | 日期时间（带时区） |
| **INTERVAL** | 16 字节 | 时间间隔 |
| **UUID** | 16 字节 | 通用唯一标识符 |
| **JSON / JSONB** | 可变 | JSON 数据 |
| **ARRAY** | 可变 | 数组类型 |
| **BYTEA** | 可变 | 二进制数据 |

## 六、Python 代码示例

```python
import psycopg2
from psycopg2.extras import RealDictCursor

# 连接数据库
conn = psycopg2.connect(
    host="localhost",
    port=5432,
    dbname="mydb",
    user="myuser",
    password="mypassword"
)

try:
    with conn.cursor(cursor_factory=RealDictCursor) as cur:
        # 创建表
        cur.execute("""
            CREATE TABLE IF NOT EXISTS products (
                id SERIAL PRIMARY KEY,
                name VARCHAR(100) NOT NULL,
                price NUMERIC(10,2) NOT NULL
            )
        """)
        
        # 插入数据
        cur.execute(
            "INSERT INTO products (name, price) VALUES (%s, %s) RETURNING id",
            ("笔记本电脑", 6999.00)
        )
        product_id = cur.fetchone()['id']
        print(f"插入成功，ID: {product_id}")
        
        # 查询数据
        cur.execute("SELECT * FROM products WHERE price > %s", (5000,))
        products = cur.fetchall()
        for p in products:
            print(f"{p['name']}: {p['price']}元")
        
        conn.commit()
        
except Exception as e:
    conn.rollback()
    print(f"错误: {e}")
finally:
    conn.close()
```

## 七、Go 代码示例

```go
package main

import (
    "database/sql"
    "fmt"
    "log"
    _ "github.com/lib/pq"
)

func main() {
    connStr := "host=localhost port=5432 user=myuser password=mypassword dbname=mydb sslmode=disable"
    db, err := sql.Open("postgres", connStr)
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()

    err = db.Ping()
    if err != nil {
        log.Fatal("无法连接数据库:", err)
    }
    fmt.Println("成功连接到 PostgreSQL")

    // 查询
    rows, err := db.Query("SELECT id, username, email FROM users WHERE is_active = $1", true)
    if err != nil {
        log.Fatal(err)
    }
    defer rows.Close()

    for rows.Next() {
        var id int
        var username, email string
        if err := rows.Scan(&id, &username, &email); err != nil {
            log.Fatal(err)
        }
        fmt.Printf("ID: %d, 用户名: %s, 邮箱: %s\n", id, username, email)
    }
}
```
