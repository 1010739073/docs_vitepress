# PostgreSQL 常见问题处理

## 一、连接问题

### 1.1 无法连接数据库

```bash
# 检查 PostgreSQL 服务状态
sudo systemctl status postgresql

# 检查端口监听
sudo netstat -tlnp | grep 5432
# 或
sudo ss -tlnp | grep 5432

# 检查进程
ps aux | grep postgres
```

**常见原因及解决方案：**

| 问题 | 原因 | 解决方案 |
|-----|------|---------|
| 连接拒绝 | 服务未启动 | `sudo systemctl start postgresql` |
| 连接超时 | 防火墙阻止 | `sudo ufw allow 5432/tcp` |
| 认证失败 | pg_hba.conf 配置 | 修改认证方法 |
| 数据库不存在 | 数据库名称错误 | `createdb mydb` |

```sql
-- 检查 pg_hba.conf 配置
-- 位置：/etc/postgresql/16/main/pg_hba.conf

-- 允许本地连接
local   all             all                                     trust
host    all             all             127.0.0.1/32            trust
host    all             all             ::1/128                 trust

-- 允许远程连接（谨慎使用）
host    all             all             0.0.0.0/0               scram-sha-256

-- 重新加载配置
SELECT pg_reload_conf();
-- 或
sudo systemctl reload postgresql
```

### 1.2 连接数过多

```sql
-- 查看当前连接数
SELECT count(*) FROM pg_stat_activity;

-- 查看连接详情
SELECT 
    datname,
    usename,
    application_name,
    client_addr,
    state,
    COUNT(*)
FROM pg_stat_activity
GROUP BY datname, usename, application_name, client_addr, state;

-- 查看最大连接数
SHOW max_connections;

-- 查看连接状态分布
SELECT state, COUNT(*) FROM pg_stat_activity GROUP BY state;

-- 终止空闲连接
SELECT pg_terminate_backend(pid) 
FROM pg_stat_activity 
WHERE state = 'idle' 
AND state_change < NOW() - INTERVAL '1 hour';

-- 终止特定连接
SELECT pg_terminate_backend(12345);
```

**解决方案：**

```sql
-- 1. 增加最大连接数（postgresql.conf）
max_connections = 200

-- 2. 使用连接池（PgBouncer）
-- 安装 PgBouncer
sudo apt-get install pgbouncer

-- 配置 pgbouncer.ini
databases]
mydb = host=localhost port=5432 dbname=mydb

[pgbouncer]
listen_port = 6432
listen_addr = *
auth_type = md5
auth_file = /etc/pgbouncer/userlist.txt
max_client_conn = 1000
default_pool_size = 20

-- 3. 优化应用程序连接使用
-- 使用连接池（HikariCP、pgbouncer 等）
-- 及时关闭不用的连接
```

## 二、性能问题

### 2.1 查询缓慢

```sql
-- 查看慢查询
SELECT 
    pid,
    now() - query_start AS duration,
    query
FROM pg_stat_activity
WHERE state != 'idle'
AND now() - query_start > interval '5 seconds'
ORDER BY duration DESC;

-- 查看 pg_stat_statements（需要安装）
SELECT 
    query,
    calls,
    total_time,
    mean_time,
    rows,
    100.0 * shared_blks_hit / nullif(shared_blks_hit + shared_blks_read, 0) AS hit_percent
FROM pg_stat_statements
ORDER BY mean_time DESC
LIMIT 10;

-- 分析查询执行计划
EXPLAIN (ANALYZE, BUFFERS, FORMAT TEXT)
SELECT * FROM users WHERE email = 'test@example.com';

-- 常见优化方法：
-- 1. 添加索引
CREATE INDEX idx_users_email ON users(email);

-- 2. 更新统计信息
ANALYZE users;

-- 3. 重写查询
-- 使用 JOIN 代替子查询
-- 避免在索引列上使用函数
-- 使用 LIMIT 代替 OFFSET（大数据量时）
```

### 2.2 表膨胀（Table Bloat）

```sql
-- 查看表膨胀情况
SELECT 
    schemaname,
    tablename,
    n_live_tup,
    n_dead_tup,
    round(n_dead_tup::numeric / nullif(n_live_tup, 0) * 100, 2) AS dead_tuple_ratio,
    pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) AS total_size
FROM pg_stat_user_tables
WHERE n_dead_tup > 1000
ORDER BY dead_tuple_ratio DESC;

-- 查看表实际大小与预期大小
SELECT 
    schemaname,
    tablename,
    pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) AS total_size,
    pg_size_pretty(pg_relation_size(schemaname||'.'||tablename)) AS table_size,
    pg_size_pretty(pg_indexes_size(schemaname||'.'||tablename)) AS index_size
FROM pg_stat_user_tables
ORDER BY pg_total_relation_size(schemaname||'.'||tablename) DESC;

-- 解决方案：
-- 1. 手动清理
VACUUM ANALYZE users;

-- 2. 完全清理（会锁表）
VACUUM FULL users;

-- 3. 调整 autovacuum 参数
ALTER TABLE users SET (
    autovacuum_vacuum_scale_factor = 0.1,
    autovacuum_analyze_scale_factor = 0.05,
    autovacuum_vacuum_threshold = 50
);

-- 4. 定期重建表（低峰期执行）
-- 使用 pg_repack 工具（不锁表）
pg_repack -d mydb -t users
```

### 2.3 索引膨胀

```sql
-- 查看索引大小和扫描次数
SELECT 
    schemaname,
    tablename,
    indexrelname,
    idx_scan,
    idx_tup_read,
    pg_size_pretty(pg_relation_size(indexrelid)) AS index_size
FROM pg_stat_user_indexes
ORDER BY pg_relation_size(indexrelid) DESC;

-- 查看索引膨胀（使用 pgstattuple 扩展）
CREATE EXTENSION IF NOT EXISTS pgstattuple;

SELECT * FROM pgstatindex('idx_users_email');

-- 重建索引
REINDEX INDEX idx_users_email;

-- 并发重建（不锁表，PostgreSQL 12+）
REINDEX INDEX CONCURRENTLY idx_users_email;

-- 删除未使用的索引
SELECT 
    schemaname,
    tablename,
    indexrelname,
    idx_scan
FROM pg_stat_user_indexes
WHERE idx_scan = 0
AND indexrelname NOT LIKE 'pg_toast%'
ORDER BY pg_relation_size(indexrelid) DESC;
```

### 2.4 内存不足

```sql
-- 查看内存设置
SHOW shared_buffers;
```

## 三、锁问题

### 3.1 锁等待

```sql
-- 查看锁等待
SELECT 
    blocked_locks.pid AS blocked_pid,
    blocked_activity.usename AS blocked_user,
    blocking_locks.pid AS blocking_pid,
    blocking_activity.usename AS blocking_user,
    blocked_activity.query AS blocked_statement,
    blocking_activity.query AS blocking_statement,
    blocked_activity.application_name AS blocked_application,
    blocking_activity.application_name AS blocking_application
FROM pg_locks blocked_locks
JOIN pg_stat_activity blocked_activity ON blocked_locks.pid = blocked_activity.pid
JOIN pg_locks blocking_locks ON blocked_locks.locktype = blocking_locks.locktype
    AND blocked_locks.relation = blocking_locks.relation
    AND blocked_locks.pid != blocking_locks.pid
JOIN pg_stat_activity blocking_activity ON blocking_locks.pid = blocking_activity.pid
WHERE NOT blocked_locks.granted;

-- 查看锁详情
SELECT 
    l.locktype,
    l.relation::regclass,
    l.mode,
    l.granted,
    a.pid,
    a.query,
    a.state
FROM pg_locks l
JOIN pg_stat_activity a ON l.pid = a.pid
WHERE l.relation IS NOT NULL
ORDER BY l.relation::regclass::text;

-- 终止阻塞进程
SELECT pg_terminate_backend(blocking_pid) FROM (
    SELECT DISTINCT blocking_locks.pid AS blocking_pid
    FROM pg_locks blocked_locks
    JOIN pg_locks blocking_locks ON blocked_locks.locktype = blocking_locks.locktype
        AND blocked_locks.relation = blocking_locks.relation
    WHERE NOT blocked_locks.granted
) t;
```

### 3.2 死锁

```sql
-- 查看死锁日志
-- 在 postgresql.conf 中设置：
-- log_lock_waits = on
-- deadlock_timeout = 1s

-- 查看日志文件
tail -f /var/log/postgresql/postgresql-16-main.log | grep -i deadlock

-- 预防死锁的方法：
-- 1. 按相同顺序访问资源
-- 2. 保持事务简短
-- 3. 使用 NOWAIT
BEGIN;
SELECT * FROM accounts WHERE id = 1 FOR UPDATE NOWAIT;
-- 如果无法立即获得锁，立即报错

-- 4. 使用 SKIP LOCKED（PostgreSQL 9.5+）
SELECT * FROM tasks WHERE status = 'pending' 
FOR UPDATE SKIP LOCKED 
LIMIT 1;
```

## 四、复制问题

### 4.1 流复制延迟

```sql
-- 在主服务器上查看复制状态
SELECT 
    application_name,
    client_addr,
    state,
    sync_state,
    sent_lsn,
    write_lsn,
    flush_lsn,
    replay_lsn,
    pg_size_pretty(pg_wal_lsn_diff(sent_lsn, replay_lsn)) AS replication_lag
FROM pg_stat_replication;

-- 在备服务器上查看接收状态
SELECT 
    status,
    receive_start_lsn,
    received_lsn,
    latest_end_lsn,
    pg_size_pretty(pg_wal_lsn_diff(received_lsn, latest_end_lsn)) AS lag
FROM pg_stat_wal_receiver;

-- 解决方案：
-- 1. 增加 wal_keep_size
-- 2. 使用复制槽
SELECT * FROM pg_replication_slots;

-- 3. 检查网络延迟
-- 4. 检查备服务器性能
-- 5. 调整 max_wal_size 和 checkpoint_timeout
```

### 4.2 复制中断

```bash
# 检查 WAL 文件
ls -la /var/lib/postgresql/16/main/pg_wal/

# 检查归档状态
psql -c "SELECT * FROM pg_stat_archiver;"

# 重新同步备服务器
# 1. 停止备服务器
sudo systemctl stop postgresql

# 2. 清理数据目录
rm -rf /var/lib/postgresql/16/main/*

# 3. 重新执行基础备份
pg_basebackup -h primary_host -U replicator -D /var/lib/postgresql/16/main -Fp -Xs -P -R

# 4. 启动备服务器
sudo systemctl start postgresql
```

## 六、数据库损坏

### 6.1 数据页损坏

```sql
-- 检查数据库一致性
-- 单表检查
SELECT pg_relation_check('users');

-- 使用 pg_dump 检查
pg_dump -d mydb -t users > /dev/null

-- 使用 amcheck 扩展检查 B-tree 索引
CREATE EXTENSION IF NOT EXISTS amcheck;

SELECT bt_index_check('idx_users_email');

-- 修复方法：
-- 1. 从备份恢复
-- 2. 使用零损坏页面（zero_damaged_pages）
SET zero_damaged_pages = on;
SELECT * FROM corrupted_table;

-- 3. 重建表
CREATE TABLE users_new AS SELECT * FROM users;
DROP TABLE users;
ALTER TABLE users_new RENAME TO users;
-- 重建索引和约束
```

### 6.2 索引损坏

```sql
-- 检查索引
REINDEX INDEX idx_users_email;

-- 并发重建（不锁表）
REINDEX INDEX CONCURRENTLY idx_users_email;

-- 重建表上所有索引
REINDEX TABLE users;

-- 并发重建表上所有索引
REINDEX TABLE CONCURRENTLY users;

-- 如果索引损坏严重，删除并重建
DROP INDEX idx_users_email;
CREATE INDEX idx_users_email ON users(email);
```

## 七、日志和监控

### 7.1 日志配置

```bash
# postgresql.conf 日志配置
logging_collector = on
log_directory = 'log'
log_filename = 'postgresql-%Y-%m-%d_%H%M%S.log'
log_rotation_age = 1d
log_rotation_size = 100MB
log_min_duration_statement = 1000  # 记录超过1秒的查询
log_checkpoints = on
log_connections = on
log_disconnections = on
log_lock_waits = on
log_temp_files = 0  # 记录所有临时文件
log_autovacuum_min_duration = 0  # 记录所有 autovacuum
```

### 7.2 常用监控查询

```sql
-- 数据库大小
SELECT 
    datname,
    pg_size_pretty(pg_database_size(datname)) AS size
FROM pg_database
ORDER BY pg_database_size(datname) DESC;

-- 表大小
SELECT 
    schemaname,
    tablename,
    pg_size_pretty(pg_total_relation_size(schemaname||'.'||tablename)) AS total_size,
    pg_size_pretty(pg_relation_size(schemaname||'.'||tablename)) AS table_size,
    pg_size_pretty(pg_indexes_size(schemaname||'.'||tablename)) AS index_size
FROM pg_tables
WHERE schemaname NOT IN ('pg_catalog', 'information_schema')
ORDER BY pg_total_relation_size(schemaname||'.'||tablename) DESC
LIMIT 20;

-- 缓存命中率
SELECT 
    sum(heap_blks_hit) / (sum(heap_blks_hit) + sum(heap_blks_read)) AS cache_hit_ratio
FROM pg_statio_user_tables;

-- 索引使用情况
SELECT 
    schemaname,
    tablename,
    indexrelname,
    idx_scan,
    idx_tup_read,
    idx_tup_fetch
FROM pg_stat_user_indexes
ORDER BY idx_scan DESC
LIMIT 20;

-- 慢查询（需要 pg_stat_statements）
SELECT 
    query,
    calls,
    total_time,
    mean_time,
    rows
FROM pg_stat_statements
ORDER BY mean_time DESC
LIMIT 10;
```

### 8.3 紧急恢复步骤

```bash
# 1. 停止 PostgreSQL
sudo systemctl stop postgresql

# 2. 备份当前数据目录（如果可能）
cp -r /var/lib/postgresql/16/main /var/lib/postgresql/16/main_backup

# 3. 从备份恢复
# 逻辑备份恢复
pg_restore -d mydb /backup/mydb_latest.dump

# 或物理备份恢复
rm -rf /var/lib/postgresql/16/main/*
pg_basebackup -h backup_server -U replicator -D /var/lib/postgresql/16/main -Fp -Xs -P

# 4. 启动 PostgreSQL
sudo systemctl start postgresql

# 5. 验证数据
psql -d mydb -c "SELECT count(*) FROM important_table;"
```
