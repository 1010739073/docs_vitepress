# 4. 索引与查询

## 索引操作

### 创建索引

```json
PUT /products
{
  "settings": {
    "number_of_shards": 5,
    "number_of_replicas": 1,
    "analysis": {
      "analyzer": {
        "my_analyzer": {
          "tokenizer": "standard",
          "filter": ["lowercase", "stop"]
        }
      }
    }
  },
  "mappings": {
    "properties": {
      "title": {
        "type": "text",
        "analyzer": "my_analyzer",
        "fields": {
          "keyword": {
            "type": "keyword"
          }
        }
      },
      "price": {
        "type": "float"
      },
      "category": {
        "type": "keyword"
      },
      "created_at": {
        "type": "date",
        "format": "yyyy-MM-dd HH:mm:ss"
      }
    }
  }
}
```

### 删除索引

```json
DELETE /products
```

### 更新索引设置

```json
PUT /products/_settings
{
  "number_of_replicas": 2
}
```

## 文档操作

### 插入文档

```json
// 指定ID
PUT /products/_doc/1
{
  "title": "iPhone 15 Pro",
  "price": 7999,
  "category": "手机",
  "created_at": "2023-09-22 10:30:00"
}

// 自动生成ID
POST /products/_doc
{
  "title": "MacBook Pro",
  "price": 12999,
  "category": "电脑",
  "created_at": "2023-01-15 09:00:00"
}
```

### 查询文档

```json
// 根据ID查询
GET /products/_doc/1

// 查询所有
GET /products/_search
{
  "query": {
    "match_all": {}
  }
}
```

### 更新文档

```json
// 全量更新
PUT /products/_doc/1
{
  "title": "iPhone 15 Pro Max",
  "price": 9999,
  "category": "手机",
  "created_at": "2023-09-22 10:30:00"
}

// 部分更新
POST /products/_update/1
{
  "doc": {
    "price": 8999
  }
}

// 脚本更新
POST /products/_update/1
{
  "script": {
    "source": "ctx._source.price += params.delta",
    "params": {
      "delta": 100
    }
  }
}
```

### 删除文档

```json
DELETE /products/_doc/1
```

## 查询语法

### 全文查询

```json
// match 查询
GET /products/_search
{
  "query": {
    "match": {
      "title": "iPhone"
    }
  }
}

// match_phrase 查询
GET /products/_search
{
  "query": {
    "match_phrase": {
      "title": "iPhone Pro"
    }
  }
}

// multi_match 查询
GET /products/_search
{
  "query": {
    "multi_match": {
      "query": "iPhone",
      "fields": ["title", "category"]
    }
  }
}
```

### 精确查询

```json
// term 查询
GET /products/_search
{
  "query": {
    "term": {
      "category": "手机"
    }
  }
}

// terms 查询
GET /products/_search
{
  "query": {
    "terms": {
      "category": ["手机", "电脑"]
    }
  }
}

// range 查询
GET /products/_search
{
  "query": {
    "range": {
      "price": {
        "gte": 5000,
        "lte": 10000
      }
    }
  }
}
```

### 复合查询

```json
// bool 查询
GET /products/_search
{
  "query": {
    "bool": {
      "must": [
        { "match": { "title": "iPhone" } }
      ],
      "filter": [
        { "range": { "price": { "lte": 8000 } } }
      ],
      "should": [
        { "term": { "category": "手机" } }
      ],
      "must_not": [
        { "term": { "category": "二手" } }
      ]
    }
  }
}

// constant_score 查询
GET /products/_search
{
  "query": {
    "constant_score": {
      "filter": {
        "term": { "category": "手机" }
      },
      "boost": 2
    }
  }
}
```

### 模糊查询

```json
// fuzzy 查询
GET /products/_search
{
  "query": {
    "fuzzy": {
      "title": {
        "value": "iphoen",
        "fuzziness": 2
      }
    }
  }
}

// wildcard 查询
GET /products/_search
{
  "query": {
    "wildcard": {
      "title": "iPh*"
    }
  }
}

// regexp 查询
GET /products/_search
{
  "query": {
    "regexp": {
      "title": "iPh.*"
    }
  }
}
```

## 聚合查询

### 基础聚合

```json
// terms 聚合
GET /sales/_search
{
  "size": 0,
  "aggs": {
    "sales_by_category": {
      "terms": {
        "field": "category",
        "size": 10
      }
    }
  }
}

// 数值聚合
GET /sales/_search
{
  "size": 0,
  "aggs": {
    "avg_price": { "avg": { "field": "price" } },
    "max_price": { "max": { "field": "price" } },
    "min_price": { "min": { "field": "price" } },
    "sum_price": { "sum": { "field": "price" } },
    "count": { "value_count": { "field": "price" } }
  }
}
```

### 嵌套聚合

```json
GET /sales/_search
{
  "size": 0,
  "aggs": {
    "by_category": {
      "terms": { "field": "category" },
      "aggs": {
        "by_date": {
          "date_histogram": {
            "field": "created_at",
            "calendar_interval": "month"
          },
          "aggs": {
            "avg_price": { "avg": { "field": "price" } }
          }
        }
      }
    }
  }
}
```

### 管道聚合

```json
GET /sales/_search
{
  "size": 0,
  "aggs": {
    "sales": {
      "date_histogram": {
        "field": "created_at",
        "calendar_interval": "day"
      },
      "aggs": {
        "total": { "sum": { "field": "price" } }
      }
    },
    "moving_avg": {
      "moving_avg": {
        "buckets_path": "sales>total",
        "window": 7
      }
    }
  }
}
```

## 分页与排序

### 分页

```json
GET /products/_search
{
  "from": 0,
  "size": 10,
  "query": {
    "match_all": {}
  }
}
```

### 排序

```json
GET /products/_search
{
  "query": {
    "match_all": {}
  },
  "sort": [
    { "price": { "order": "desc" } },
    { "created_at": { "order": "asc" } }
  ]
}
```

### 深分页

```json
// 使用 search_after
GET /products/_search
{
  "size": 10000,
  "query": { "match_all": {} },
  "sort": [
    { "_id": "asc" }
  ],
  "search_after": ["last_id"]
}
```
