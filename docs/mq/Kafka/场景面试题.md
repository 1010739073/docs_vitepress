# Kafka 场景面试题

## 场景一：秒杀系统

### 问题

设计一个秒杀系统，如何使用 Kafka 保证高并发下的数据一致性？

### 解决方案

**1. 整体架构**

```
用户请求 → API Gateway → Redis 预扣减 → Kafka 异步处理 → MySQL 落库
                    ↓
               直接返回成功
```

**2. Redis 预扣减**

```java
@Service
public class SeckillService {
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    @Autowired
    private KafkaTemplate<String, String> kafkaTemplate;

    public Result placeOrder(Long userId, Long productId) {
        // 1. 检查库存
        String key = "seckill:stock:" + productId;
        Long remaining = redisTemplate.opsForValue().decrement(key);

        if (remaining == null || remaining < 0) {
            // 库存不足，回滚
            redisTemplate.opsForValue().increment(key);
            return Result.fail("库存不足");
        }

        // 2. 检查重复购买
        String purchaseKey = "seckill:purchase:" + userId + ":" + productId;
        Boolean isNew = redisTemplate.opsForValue().setIfAbsent(purchaseKey, "1", 86400);

        if (!isNew) {
            // 回滚库存
            redisTemplate.opsForValue().increment(key);
            return Result.fail("不能重复购买");
        }

        // 3. 发送消息到 Kafka
        SeckillOrder order = new SeckillOrder(userId, productId, System.currentTimeMillis());
        kafkaTemplate.send("seckill-topic", order.toString());

        // 4. 直接返回成功
        return Result.success("秒杀成功");
    }
}
```

**3. Kafka 消费者**

```java
@Service
public class SeckillConsumer {
    @Autowired
    private OrderService orderService;
    @Autowired
    private RedisTemplate<String, String> redisTemplate;

    @KafkaListener(topics = "seckill-topic", groupId = "seckill-group",
        concurrency = "10")
    public void consumeSeckillOrder(String message) {
        try {
            SeckillOrder order = JSON.parseObject(message, SeckillOrder.class);

            // 1. 创建订单
            Order dbOrder = new Order();
            dbOrder.setUserId(order.getUserId());
            dbOrder.setProductId(order.getProductId());
            dbOrder.setOrderTime(new Date(order.getTimestamp()));

            // 2. 扣减数据库库存
            int rows = orderService.createOrder(dbOrder);

            if (rows > 0) {
                // 3. 标记完成
                redisTemplate.opsForValue().set(
                    "seckill:complete:" + order.getUserId() + ":" + order.getProductId(),
                    "1", 86400
                );
            } else {
                // 库存不足，回滚
                redisTemplate.opsForValue().increment("seckill:stock:" + order.getProductId());
            }
        } catch (Exception e) {
            // 记录日志，后续补偿
            log.error("处理秒杀订单失败: {}", message, e);
            // 放入延迟队列
            redisTemplate.opsForList().leftPush("seckill:retry", message);
        }
    }
}
```

**4. 补偿机制**

```java
@Service
public class SeckillCompensator {
    @Autowired
    private RedisTemplate<String, String> redisTemplate;
    @Autowired
    private OrderService orderService;

    @Scheduled(fixedDelay = 5000)
    public void compensate() {
        // 1. 检查超时未完成的订单
        String pattern = "seckill:purchase:*";
        Set<String> keys = redisTemplate.keys(pattern);

        for (String key : keys) {
            String value = redisTemplate.opsForValue().get(key);
            if ("1".equals(value)) {
                // 2. 检查是否在数据库中
                String completeKey = key.replace("purchase", "complete");
                if (!redisTemplate.hasKey(completeKey)) {
                    // 3. 重新处理
                    String[] parts = key.split(":");
                    Long userId = Long.parseLong(parts[2]);
                    Long productId = Long.parseLong(parts[3]);

                    // 重新创建订单
                    Order order = new Order();
                    order.setUserId(userId);
                    order.setProductId(productId);
                    orderService.createOrder(order);

                    // 标记完成
                    redisTemplate.opsForValue().set(completeKey, "1", 86400);
                }
            }
        }
    }
}
```

```java
        }
    }
```

**6. 幂等性处理**

```java
@Service
public class IdempotentProcessor {
    @Autowired
    private RedisTemplate<String, String> redisTemplate;

    public boolean processWithIdempotency(String key, Runnable action) {
        String lockKey = "lock:" + key;

        // 使用 Redis 实现分布式锁
        Boolean locked = redisTemplate.opsForValue().setIfAbsent(
            lockKey, "1", 300);

        if (!locked) {
            log.info("正在处理: {}", key);
            return false;
        }

        try {
            // 检查是否已处理
            String processedKey = "processed:" + key;
            if (redisTemplate.hasKey(processedKey)) {
                log.info("已处理: {}", key);
                return false;
            }

            // 执行处理
            action.run();

            // 标记已处理
            redisTemplate.opsForValue().set(processedKey, "1", 86400);
            return true;
        } finally {
            redisTemplate.delete(lockKey);
        }
    }
}

// 使用示例
@Service
public class LogisticsConsumer {
    @Autowired
    private IdempotentProcessor idempotentProcessor;

    public void consumeOrderStatus(String message) {
        OrderStatusEvent event = JSON.parseObject(message, OrderStatusEvent.class);
        String key = event.getOrderId() + ":" + event.getStatus();

        idempotentProcessor.processWithIdempotency(key, () -> {
            // 处理逻辑
            logisticsService.processOrder(event.getOrderId());
        });
    }
}
```

**3. Flink 实时计算**

```java
public class RealtimeMetricsJob {
    public static void main(String[] args) throws Exception {
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();

        // 1. 消费用户事件
        FlinkKafkaConsumer<UserEvent> consumer = new FlinkKafkaConsumer<>(
            "user-events",
            new JsonDeserializationSchema<>(UserEvent.class),
            kafkaProps
        );

        DataStream<UserEvent> events = env.addSource(consumer)
            .assignTimestampsAndWatermarks(
                WatermarkStrategy.<UserEvent>forBoundedOutOfOrderness(Duration.ofSeconds(5))
                    .withTimestampAssigner((event, timestamp) -> event.getTimestamp())
            );

        // 2. 计算订单量（1 分钟窗口）
        DataStream<Metric> orderCount = events
            .filter(event -> "ORDER".equals(event.getType()))
            .keyBy(UserEvent::getEventType)
            .window(TumblingEventTimeWindows.of(Time.minutes(1)))
            .aggregate(new OrderCountAggregator());

        // 3. 计算交易额（1 分钟窗口）
        DataStream<Metric> orderAmount = events
            .filter(event -> "ORDER".equals(event.getType()))
            .keyBy(UserEvent::getEventType)
            .window(TumblingEventTimeWindows.of(Time.minutes(1)))
            .aggregate(new OrderAmountAggregator());

        // 4. 计算用户数（1 分钟窗口）
        DataStream<Metric> userCount = events
            .keyBy(UserEvent::getUserId)
            .window(TumblingEventTimeWindows.of(Time.minutes(1)))
            .process(new UserCountProcessFunction());

        // 5. 合并所有指标
        DataStream<Metric> allMetrics = orderCount
            .union(orderAmount, userCount);

        // 6. 发送到 Redis
        allMetrics.addSink(new RedisSink());

        env.execute("Realtime Metrics");
    }
}
```

**4. Redis 存储**

```java
public class RedisSink extends RichSinkFunction<Metric> {
    private JedisPool jedisPool;

    @Override
    public void open(Configuration parameters) {
        jedisPool = new JedisPool("localhost", 6379);
    }

    @Override
    public void invoke(Metric metric) {
        try (Jedis jedis = jedisPool.getResource()) {
            String key = "metrics:" + metric.getName() + ":" + metric.getWindow();
            jedis.set(key, metric.getValue());
            jedis.expire(key, 3600);
        }
    }

    @Override
    public void close() {
        jedisPool.close();
    }
}
```

```java
public class FailingSink extends RichSinkFunction<Metric> {

    @Override

        props.put("bootstrap.servers", "localhost:9092");

        dlqProducer.close();
    }
}
```

```java
public class LogPartitioner implements Partitioner {
    @Override
    public int partition(String topic, Object key, byte[] keyBytes,
                         Object value, byte[] valueBytes, Cluster cluster) {
        LogEntry log = JSON.parseObject(new String(valueBytes), LogEntry.class);
        List<PartitionInfo> partitions = cluster.partitionsForTopic(topic);

        // 按 应用 + 环境 + 日志级别 分区
        String partitionKey = log.getApplication() + ":" +
                            log.getEnvironment() + ":" +
                            log.getLevel();
        int hash = Math.abs(partitionKey.hashCode());
        return hash % partitions.size();
    }
}
```

**4. Flink 实时处理**

```java
public class LogProcessingJob {
    public static void main(String[] args) throws Exception {
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();

        // 1. 消费日志
        FlinkKafkaConsumer<LogEntry> consumer = new FlinkKafkaConsumer<>(
            "application-logs",
            new JsonDeserializationSchema<>(LogEntry.class),
            kafkaProps
        );

        DataStream<LogEntry> logs = env.addSource(consumer);

        // 2. 过滤 ERROR 日志
        DataStream<LogEntry> errorLogs = logs
            .filter(log -> "ERROR".equals(log.getLevel()));

        // 3. 窗口聚合，检测异常
        DataStream<Alert> alerts = errorLogs
            .keyBy(LogEntry::getApplication)
            .window(SlidingEventTimeWindows.of(Time.minutes(5), Time.minutes(1)))
            .aggregate(new ErrorCountAggregator())
            .filter(count -> count.getCount() > 10)
            .map(count -> new Alert(
                count.getApplication(),
                "ERROR 日志过多",
                count.getCount()
            ));

        // 4. 发送告警
        FlinkKafkaProducer<Alert> alertProducer = new FlinkKafkaProducer<>(
            "log-alerts",
            new JsonSerializationSchema<>(),
            kafkaProps,
            FlinkKafkaProducer.Semantic.AT_LEAST_ONCE
        );
        alerts.addSink(alertProducer);

        // 5. 发送到 Elasticsearch
        logs.addSink(new ElasticsearchSink<>(esConfig));

        env.execute("Log Processing");
    }
}
```

```java
@Service
public class LogArchiveService {
    @Autowired
    private KafkaTemplate<String, String> kafkaTemplate;

    @Scheduled(cron = "0 0 * * * ?")

        List<LogEntry> logs = logRepository.findLogsBefore(LocalDateTime.now().minusDays(7));

}
```

```java
public class InterestCalculationJob {
    public static void main(String[] args) throws Exception {
        StreamExecutionEnvironment env = StreamExecutionEnvironment.getExecutionEnvironment();

        // 1. 消费用户行为
        FlinkKafkaConsumer<UserBehavior> consumer = new FlinkKafkaConsumer<>(
            "user-behaviors",
            new JsonDeserializationSchema<>(UserBehavior.class),
            kafkaProps
        );

        DataStream<UserBehavior> behaviors = env.addSource(consumer)
            .assignTimestampsAndWatermarks(
                WatermarkStrategy.<UserBehavior>forBoundedOutOfOrderness(Duration.ofSeconds(10))
                    .withTimestampAssigner((behavior, timestamp) -> behavior.getTimestamp())
            );

        // 2. 计算兴趣标签
        DataStream<UserInterest> interests = behaviors
            .keyBy(UserBehavior::getUserId)
            .window(SlidingEventTimeWindows.of(Time.days(30), Time.hours(1)))
            .aggregate(new InterestAggregator());

        // 3. 存储到 HBase
        interests.addSink(new HBaseSink());

        env.execute("Interest Calculation");
    }
}

public class InterestAggregator implements AggregateFunction<
    UserBehavior, Map<String, Double>, UserInterest> {

    @Override
    public Map<String, Double> createAccumulator() {
        return new HashMap<>();
    }

    @Override
    public Map<String, Double> add(UserBehavior behavior, Map<String, Double> acc) {
        String category = behavior.getCategory();
        double weight = getWeight(behavior.getAction());

        acc.put(category, acc.getOrDefault(category, 0.0) + weight);
        return acc;
    }

    @Override
    public Map<String, Double> merge(Map<String, Double> a, Map<String, Double> b) {
        b.forEach((k, v) -> a.merge(k, v, Double::sum));
        return a;
    }

    @Override
    public UserInterest getResult(Map<String, Double> acc) {
        // 归一化
        double total = acc.values().stream().mapToDouble(Double::doubleValue).sum();
        Map<String, Double> normalized = acc.entrySet().stream()
            .collect(Collectors.toMap(
                Map.Entry::getKey,
                e -> e.getValue() / total
            ));

        return new UserInterest(userId, normalized);
    }

    private double getWeight(String action) {
        switch (action) {
            case "VIEW": return 1.0;
            case "CLICK": return 3.0;
            case "LIKE": return 5.0;
            case "PURCHASE": return 10.0;
            default: return 0.0;
        }
    }
}
```

```java
public class BehaviorSequenceService {
    @Autowired
    private RedisTemplate<String, String> redisTemplate;

    public void recordBehavior(UserBehavior behavior) {
        String key = "behavior:sequence:" + behavior.getUserId();

        // 使用 List 存储行为序列
        String value = behavior.getCategory() + ":" + behavior.getAction() +
                      ":" + behavior.getTimestamp();
        redisTemplate.opsForList().rightPush(key, value);

        // 保留最近 100 条行为
        redisTemplate.opsForList().trim(key, -100, -1);

        // 设置过期时间
        redisTemplate.expire(key, 7, TimeUnit.DAYS);
    }

    public List<String> getRecentBehavior(Long userId, int count) {
        String key = "behavior:sequence:" + userId;
        Long size = redisTemplate.opsForList().size(key);
        if (size == null || size == 0) {
            return Collections.emptyList();
        }

        return redisTemplate.opsForList().range(key, -count, -1);
    }
}
```

```java
@Service
public class ConflictResolver {
    @Autowired
    private DataVersionService versionService;

    public void resolveConflict(Data local, Data remote) {
        // 1. 获取版本号
        long localVersion = versionService.getVersion(local.getId());
        long remoteVersion = remote.getVersion();

        // 2. 比较版本号
        if (localVersion > remoteVersion) {
            // 本地版本更新，覆盖远程
            updateRemote(local);
        } else if (remoteVersion > localVersion) {
            // 远程版本更新，覆盖本地
            updateLocal(remote);
        } else {
            // 版本号相同，使用时间戳
            if (local.getTimestamp() > remote.getTimestamp()) {
                updateRemote(local);
            } else {
                updateLocal(remote);
            }
        }
    }
}
```

```java
@Service
public class ConsistencyChecker {
    @Autowired
    private LocalDataService localDataService;
    @Autowired
    private RemoteDataService remoteDataService;

    @Scheduled(fixedDelay = 3600000)
    public void checkConsistency() {
        // 1. 获取本地数据
        List<Data> localData = localDataService.getAll();

        // 2. 获取远程数据
        List<Data> remoteData = remoteDataService.getAll();

        // 3. 比较数据
        Map<String, Data> remoteMap = remoteData.stream()
            .collect(Collectors.toMap(Data::getId, Function.identity()));

        List<String> inconsistencies = new ArrayList<>();

        for (Data local : localData) {
            Data remote = remoteMap.get(local.getId());
            if (remote == null) {
                inconsistencies.add("本地数据在远程不存在: " + local.getId());
            } else if (!local.equals(remote)) {
                inconsistencies.add("数据不一致: " + local.getId());
            }
        }

        // 4. 记录不一致
        if (!inconsistencies.isEmpty()) {
            log.warn("数据不一致: {}", inconsistencies);
            alertService.sendAlert("数据不一致", inconsistencies);
        }
    }
}
```

**5. 故障切换**

```java
@Service
public class FailoverService {
    @Value("${cluster.mode}")
    private String clusterMode;  // primary 或 backup

    @Autowired
    private HealthChecker healthChecker;

    public void handleFailover() {
        // 1. 检查健康状态
        if (healthChecker.isHealthy()) {
            return;
        }

        // 2. 切换模式
        if ("primary".equals(clusterMode)) {
            // Primary 故障，切换到 Backup
            switchToBackup();
        } else {
            // Backup 故障，尝试切换到 Primary
            switchToPrimary();
        }
    }

    private void switchToBackup() {
        log.info("切换到 Backup 模式");
        clusterMode = "backup";

        // 1. 停止写操作
        stopWriteOperations();

        // 2. 等待数据同步
        waitForDataSync();

        // 3. 切换读操作
        switchReadOperations();

        // 4. 发送告警
        alertService.sendAlert("已切换到 Backup 模式");
    }

    private void switchToPrimary() {
        log.info("尝试切换到 Primary 模式");

        // 1. 检查 Primary 是否恢复
        if (!healthChecker.isPrimaryHealthy()) {
            return;
        }

        // 2. 同步数据到 Primary
        syncDataToPrimary();

        // 3. 切换模式
        clusterMode = "primary";

        // 4. 恢复写操作
        startWriteOperations();

        // 5. 切换读操作
        switchReadOperations();

        // 6. 发送通知
        alertService.sendAlert("已恢复到 Primary 模式");
    }
}
```
