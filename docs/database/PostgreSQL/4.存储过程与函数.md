# PostgreSQL 存储过程与函数

## 一、函数基础

### 1.1 创建函数

```sql
-- 基本函数结构
CREATE OR REPLACE FUNCTION function_name(param1 TYPE, param2 TYPE)
RETURNS return_type AS $$
BEGIN
    -- 函数体
    RETURN result;
END;
$$ LANGUAGE plpgsql;

-- 简单函数示例
CREATE OR REPLACE FUNCTION get_user_count()
RETURNS INTEGER AS $$
DECLARE
    count INTEGER;
BEGIN
    SELECT COUNT(*) INTO count FROM users;
    RETURN count;
END;
$$ LANGUAGE plpgsql;

-- 调用函数
SELECT get_user_count();
```

### 1.2 函数参数模式

```sql
-- IN 参数（默认）
CREATE OR REPLACE FUNCTION greet_user(user_name VARCHAR)
RETURNS TEXT AS $$
BEGIN
    RETURN 'Hello, ' || user_name;
END;
$$ LANGUAGE plpgsql;

-- OUT 参数
CREATE OR REPLACE FUNCTION get_user_stats(OUT total INT, OUT active INT)
AS $$
BEGIN
    SELECT COUNT(*) INTO total FROM users;
    SELECT COUNT(*) INTO active FROM users WHERE is_active = TRUE;
END;
$$ LANGUAGE plpgsql;

-- 调用
SELECT * FROM get_user_stats();

-- INOUT 参数
CREATE OR REPLACE FUNCTION increment(INOUT val INT)
AS $$
BEGIN
    val := val + 1;
END;
$$ LANGUAGE plpgsql;

-- 调用
SELECT increment(5);  -- 返回 6

-- 多个 OUT 参数
CREATE OR REPLACE FUNCTION get_user_by_id(
    IN user_id INT,
    OUT user_name VARCHAR,
    OUT user_email VARCHAR,
    OUT user_age INT
)
AS $$
BEGIN
    SELECT username, email, age
    INTO user_name, user_email, user_age
    FROM users
    WHERE id = user_id;
END;
$$ LANGUAGE plpgsql;
```

### 1.3 返回类型

```sql
-- 返回标量
CREATE OR REPLACE FUNCTION get_user_name(user_id INT)
RETURNS VARCHAR AS $$
DECLARE
    name VARCHAR;
BEGIN
    SELECT username INTO name FROM users WHERE id = user_id;
    RETURN name;
END;
$$ LANGUAGE plpgsql;

-- 返回复合类型（表行）
CREATE OR REPLACE FUNCTION get_user(user_id INT)
RETURNS users AS $$
DECLARE
    user_record users%ROWTYPE;
BEGIN
    SELECT * INTO user_record FROM users WHERE id = user_id;
    RETURN user_record;
END;
$$ LANGUAGE plpgsql;

-- 调用
SELECT * FROM get_user(1);

-- 返回表
CREATE OR REPLACE FUNCTION get_active_users()
RETURNS TABLE (id INT, username VARCHAR, email VARCHAR) AS $$
BEGIN
    RETURN QUERY
    SELECT u.id, u.username, u.email
    FROM users u
    WHERE u.is_active = TRUE;
END;
$$ LANGUAGE plpgsql;

-- 调用
SELECT * FROM get_active_users();

-- 返回 SETOF
CREATE OR REPLACE FUNCTION get_users_by_age(min_age INT)
RETURNS SETOF users AS $$
BEGIN
    RETURN QUERY
    SELECT * FROM users WHERE age >= min_age;
END;
$$ LANGUAGE plpgsql;
```

### 2.2 控制结构

```sql
-- IF 语句
CREATE OR REPLACE FUNCTION get_age_group(age INT)
RETURNS TEXT AS $$
BEGIN
    IF age < 18 THEN
        RETURN '未成年';
    ELSIF age < 60 THEN
        RETURN '成年';
    ELSE
        RETURN '老年';
    END IF;
END;
$$ LANGUAGE plpgsql;

-- CASE 语句
CREATE OR REPLACE FUNCTION get_status_text(status VARCHAR)
RETURNS TEXT AS $$
BEGIN
    CASE status
        WHEN 'active' THEN RETURN '活跃';
        WHEN 'inactive' THEN RETURN '不活跃';
        WHEN 'pending' THEN RETURN '待处理';
        ELSE RETURN '未知';
    END CASE;
END;
$$ LANGUAGE plpgsql;

-- LOOP 循环
CREATE OR REPLACE FUNCTION sum_numbers(n INT)
RETURNS INT AS $$
DECLARE
    i INT := 1;
    sum INT := 0;
BEGIN
    LOOP
        EXIT WHEN i > n;
        sum := sum + i;
        i := i + 1;
    END LOOP;
    RETURN sum;
END;
$$ LANGUAGE plpgsql;

-- WHILE 循环
CREATE OR REPLACE FUNCTION factorial(n INT)
RETURNS INT AS $$
DECLARE
    result INT := 1;
    i INT := 1;
BEGIN
    WHILE i <= n LOOP
        result := result * i;
        i := i + 1;
    END LOOP;
    RETURN result;
END;
$$ LANGUAGE plpgsql;

-- FOR 循环
CREATE OR REPLACE FUNCTION sum_array(arr INT[])
RETURNS INT AS $$
DECLARE
    total INT := 0;
    x INT;
BEGIN
    FOREACH x IN ARRAY arr LOOP
        total := total + x;
    END LOOP;
    RETURN total;
END;
$$ LANGUAGE plpgsql;

-- 查询循环
CREATE OR REPLACE FUNCTION process_users()
RETURNS VOID AS $$
DECLARE
    user_rec RECORD;
BEGIN
    FOR user_rec IN SELECT * FROM users WHERE is_active = TRUE
    LOOP
        -- 处理每个用户
        RAISE NOTICE 'Processing user: %', user_rec.username;
    END LOOP;
END;
$$ LANGUAGE plpgsql;
```

### 2.3 异常处理

```sql
CREATE OR REPLACE FUNCTION safe_divide(a NUMERIC, b NUMERIC)
RETURNS NUMERIC AS $$
BEGIN
    RETURN a / b;
EXCEPTION
    WHEN division_by_zero THEN
        RAISE NOTICE '除数不能为零';
        RETURN NULL;
    WHEN OTHERS THEN
        RAISE NOTICE '错误: %', SQLERRM;
        RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- 完整异常处理示例
CREATE OR REPLACE FUNCTION create_user(
    p_username VARCHAR,
    p_email VARCHAR,
    p_age INT
)
RETURNS INT AS $$
DECLARE
    new_id INT;
BEGIN
    INSERT INTO users (username, email, age)
    VALUES (p_username, p_email, p_age)
    RETURNING id INTO new_id;
    
    RETURN new_id;
    
EXCEPTION
    WHEN unique_violation THEN
        RAISE EXCEPTION '用户已存在: %', p_username;
    WHEN check_violation THEN
        RAISE EXCEPTION '数据验证失败';
    WHEN OTHERS THEN
        RAISE EXCEPTION '创建用户失败: %', SQLERRM;
END;
$$ LANGUAGE plpgsql;
```

## 三、存储过程

### 3.1 创建存储过程

```sql
-- PostgreSQL 11+ 支持存储过程
CREATE OR REPLACE PROCEDURE transfer_money(
    from_account INT,
    to_account INT,
    amount NUMERIC
)
AS $$
BEGIN
    -- 扣款
    UPDATE accounts 
    SET balance = balance - amount 
    WHERE id = from_account;
    
    IF NOT FOUND THEN
        RAISE EXCEPTION '账户不存在: %', from_account;
    END IF;
    
    -- 入账
    UPDATE accounts 
    SET balance = balance + amount 
    WHERE id = to_account;
    
    IF NOT FOUND THEN
        RAISE EXCEPTION '账户不存在: %', to_account;
    END IF;
    
    -- 记录日志
    INSERT INTO transaction_log (from_account, to_account, amount, created_at)
    VALUES (from_account, to_account, amount, NOW());
    
    COMMIT;
EXCEPTION
    WHEN OTHERS THEN
        ROLLBACK;
        RAISE;
END;
$$ LANGUAGE plpgsql;

-- 调用存储过程
CALL transfer_money(1, 2, 100.00);
```

### 3.2 带事务控制的存储过程

```sql
CREATE OR REPLACE PROCEDURE batch_process_orders()
AS $$
DECLARE
    order_rec RECORD;
    processed_count INT := 0;
BEGIN
    FOR order_rec IN SELECT * FROM orders WHERE status = 'pending'
    LOOP
        BEGIN
            -- 处理订单
            UPDATE orders SET status = 'processing' WHERE id = order_rec.id;
            
            -- 模拟处理
            PERFORM pg_sleep(0.1);
            
            UPDATE orders SET status = 'completed' WHERE id = order_rec.id;
            
            processed_count := processed_count + 1;
            
            -- 每10个订单提交一次
            IF processed_count % 10 = 0 THEN
                COMMIT;
            END IF;
            
        EXCEPTION
            WHEN OTHERS THEN
                -- 记录错误，继续处理下一个
                INSERT INTO error_log (order_id, error_message, created_at)
                VALUES (order_rec.id, SQLERRM, NOW());
                CONTINUE;
        END;
    END LOOP;
    
    -- 提交剩余
    COMMIT;
END;
$$ LANGUAGE plpgsql;
```

## 四、触发器

### 4.1 创建触发器

```sql
-- 触发器函数
CREATE OR REPLACE FUNCTION log_user_changes()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        INSERT INTO audit_log (table_name, operation, record_id, new_data, created_at)
        VALUES ('users', 'INSERT', NEW.id, row_to_json(NEW), NOW());
        RETURN NEW;
        
    ELSIF TG_OP = 'UPDATE' THEN
        INSERT INTO audit_log (table_name, operation, record_id, old_data, new_data, created_at)
        VALUES ('users', 'UPDATE', NEW.id, row_to_json(OLD), row_to_json(NEW), NOW());
        RETURN NEW;
        
    ELSIF TG_OP = 'DELETE' THEN
        INSERT INTO audit_log (table_name, operation, record_id, old_data, created_at)
        VALUES ('users', 'DELETE', OLD.id, row_to_json(OLD), NOW());
        RETURN OLD;
    END IF;
    
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

-- 创建触发器
CREATE TRIGGER trg_users_audit
AFTER INSERT OR UPDATE OR DELETE ON users
FOR EACH ROW
EXECUTE FUNCTION log_user_changes();
```

### 4.2 触发器类型

```sql
-- BEFORE 触发器
CREATE OR REPLACE FUNCTION set_created_at()
RETURNS TRIGGER AS $$
BEGIN
    IF TG_OP = 'INSERT' THEN
        NEW.created_at := NOW();
    END IF;
    NEW.updated_at := NOW();
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_users_timestamp
BEFORE INSERT OR UPDATE ON users
FOR EACH ROW
EXECUTE FUNCTION set_created_at();

-- INSTEAD OF 触发器（视图）
CREATE VIEW user_orders_view AS
SELECT u.id, u.username, o.id AS order_id, o.total_amount
FROM users u
LEFT JOIN orders o ON u.id = o.user_id;

CREATE OR REPLACE FUNCTION insert_user_order()
RETURNS TRIGGER AS $$
DECLARE
    new_user_id INT;
BEGIN
    -- 插入用户
    INSERT INTO users (username) VALUES (NEW.username)
    RETURNING id INTO new_user_id;
    
    -- 插入订单
    IF NEW.order_id IS NOT NULL THEN
        INSERT INTO orders (user_id, total_amount)
        VALUES (new_user_id, NEW.total_amount);
    END IF;
    
    RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_user_orders_insert
INSTEAD OF INSERT ON user_orders_view
FOR EACH ROW
EXECUTE FUNCTION insert_user_order();

-- 语句级触发器
CREATE OR REPLACE FUNCTION log_table_changes()
RETURNS TRIGGER AS $$
BEGIN
    INSERT INTO table_change_log (table_name, operation, changed_at)
    VALUES (TG_TABLE_NAME, TG_OP, NOW());
    RETURN NULL;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER trg_orders_log
AFTER INSERT OR UPDATE OR DELETE ON orders
FOR EACH STATEMENT
EXECUTE FUNCTION log_table_changes();

-- 条件触发器
CREATE TRIGGER trg_high_value_orders
AFTER INSERT ON orders
FOR EACH ROW
WHEN (NEW.total_amount > 1000)
EXECUTE FUNCTION notify_high_value_order();
```

### 4.3 触发器管理

```sql
-- 启用/禁用触发器
ALTER TABLE users DISABLE TRIGGER trg_users_audit;
ALTER TABLE users ENABLE TRIGGER trg_users_audit;

-- 禁用/启用所有触发器
ALTER TABLE users DISABLE TRIGGER ALL;
ALTER TABLE users ENABLE TRIGGER ALL;

-- 查看触发器
SELECT * FROM pg_trigger WHERE tgrelid = 'users'::regclass;

-- 删除触发器
DROP TRIGGER IF EXISTS trg_users_audit ON users;
```

## 五、高级特性

### 5.1 动态 SQL

```sql
CREATE OR REPLACE FUNCTION dynamic_query(table_name TEXT, column_name TEXT)
RETURNS TABLE (result TEXT) AS $$
BEGIN
    RETURN QUERY EXECUTE format('SELECT %I::TEXT FROM %I', column_name, table_name);
END;
$$ LANGUAGE plpgsql;

-- 带参数的动态 SQL
CREATE OR REPLACE FUNCTION search_users(search_column TEXT, search_value TEXT)
RETURNS SETOF users AS $$
BEGIN
    RETURN QUERY EXECUTE format(
        'SELECT * FROM users WHERE %I = $1',
        search_column
    ) USING search_value;
END;
$$ LANGUAGE plpgsql;

-- 安全的动态 SQL（防止 SQL 注入）
CREATE OR REPLACE FUNCTION safe_dynamic_query(
    p_table_name TEXT,
    p_column_names TEXT[],
    p_where_column TEXT,
    p_where_value TEXT
)
RETURNS SETOF RECORD AS $$
DECLARE
    sql_query TEXT;
    valid_tables TEXT[] := ARRAY['users', 'orders', 'products'];
BEGIN
    -- 验证表名
    IF NOT p_table_name = ANY(valid_tables) THEN
        RAISE EXCEPTION '无效的表名: %', p_table_name;
    END IF;
    
    -- 构建查询
    sql_query := format(
        'SELECT %s FROM %I WHERE %I = $1',
        array_to_string(p_column_names, ', '),
        p_table_name,
        p_where_column
    );
    
    RETURN QUERY EXECUTE sql_query USING p_where_value;
END;
$$ LANGUAGE plpgsql;
```

### 5.2 游标

```sql
CREATE OR REPLACE FUNCTION process_large_table()
RETURNS VOID AS $$
DECLARE
    rec RECORD;
    cur CURSOR FOR SELECT * FROM large_table;
BEGIN
    OPEN cur;
    LOOP
        FETCH cur INTO rec;
        EXIT WHEN NOT FOUND;
        
        -- 处理记录
        PERFORM process_record(rec);
        
        -- 每1000条提交一次
        IF (cur%ROWCOUNT % 1000) = 0 THEN
            COMMIT;
        END IF;
    END LOOP;
    CLOSE cur;
END;
$$ LANGUAGE plpgsql;

-- 带参数的游标
CREATE OR REPLACE FUNCTION get_users_cursor(min_age INT)
RETURNS REFCURSOR AS $$
DECLARE
    cur REFCURSOR;
BEGIN
    OPEN cur FOR SELECT * FROM users WHERE age >= min_age;
    RETURN cur;
END;
$$ LANGUAGE plpgsql;
```

## 六、最佳实践

### 6.1 函数设计原则

```sql
-- 1. 使用 OR REPLACE
CREATE OR REPLACE FUNCTION my_function()

-- 2. 明确指定参数模式
CREATE FUNCTION example(IN p1 INT, OUT p2 INT, INOUT p3 INT)

-- 3. 使用 %TYPE 和 %ROWTYPE
DECLARE
    v_user_id users.id%TYPE;
    v_user_record users%ROWTYPE;

-- 4. 处理异常
EXCEPTION
    WHEN specific_error THEN
        -- 处理特定错误
    WHEN OTHERS THEN
        -- 处理其他错误

-- 5. 使用 RAISE 记录日志
RAISE NOTICE 'Processing user: %', user_id;
RAISE WARNING 'Invalid input: %', input_value;
RAISE EXCEPTION 'Error: %', error_message;
```

### 6.2 性能优化

```sql
-- 1. 使用 SQL 函数（内联）
CREATE OR REPLACE FUNCTION get_active_user_count()
RETURNS BIGINT AS $$
    SELECT COUNT(*) FROM users WHERE is_active = TRUE;
$$ LANGUAGE SQL IMMUTABLE;

-- 2. 标记函数特性
CREATE FUNCTION stable_function() RETURNS INT
LANGUAGE SQL STABLE;  -- 不修改数据库，对相同输入返回相同结果

CREATE FUNCTION immutable_function() RETURNS INT
LANGUAGE SQL IMMUTABLE;  -- 不访问数据库，纯计算

-- 3. 避免在循环中执行 SQL
-- 不好
FOR i IN 1..1000 LOOP
    INSERT INTO logs VALUES (i, 'message');
END LOOP;

-- 好
INSERT INTO logs SELECT i, 'message' FROM generate_series(1, 1000) AS i;

-- 4. 使用 RETURN QUERY
CREATE FUNCTION get_users()
RETURNS SETOF users AS $$
BEGIN
    RETURN QUERY SELECT * FROM users WHERE is_active = TRUE;
END;
$$ LANGUAGE plpgsql;
```

### 6.3 安全建议

```sql
-- 1. 使用 SECURITY DEFINER 谨慎
CREATE FUNCTION admin_function()
RETURNS VOID
SECURITY DEFINER  -- 以创建者权限执行
AS $$
BEGIN
    -- 敏感操作
END;
$$ LANGUAGE plpgsql;

-- 2. 搜索路径安全
CREATE FUNCTION safe_function()
RETURNS VOID
AS $$
BEGIN
    -- 设置搜索路径
    SET LOCAL search_path = public, pg_temp;
    -- 操作
END;
$$ LANGUAGE plpgsql;

-- 3. 验证输入
CREATE FUNCTION process_user_input(input TEXT)
RETURNS VOID
AS $$
BEGIN
    IF input IS NULL OR length(trim(input)) = 0 THEN
        RAISE EXCEPTION '输入不能为空';
    END IF;
    -- 处理
END;
$$ LANGUAGE plpgsql;
```
