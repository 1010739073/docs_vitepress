# SQL Server 存储过程与函数

## 1. 存储过程基础

### 什么是存储过程

存储过程是一组预编译的 SQL 语句，存储在数据库中，可以通过名称调用执行。它具有以下优点：

- **性能优化**：预编译，减少网络传输
- **安全性**：可以限制对表的直接访问
- **代码复用**：多个应用程序可以共享同一存储过程
- **减少网络流量**：一次调用执行多条语句

### 创建存储过程

```sql
-- 创建简单存储过程
CREATE PROCEDURE usp_GetAllUsers
AS
BEGIN
    SELECT * FROM Users;
END;

-- 执行存储过程
EXEC usp_GetAllUsers;
```

### 带参数的存储过程

```sql
-- 创建带输入参数的存储过程
CREATE PROCEDURE usp_GetUserByID
    @UserID INT
AS
BEGIN
    SELECT * FROM Users WHERE UserID = @UserID;
END;

-- 执行带参数的存储过程
EXEC usp_GetUserByID @UserID = 1;

-- 创建带多个参数的存储过程
CREATE PROCEDURE usp_SearchUsers
    @MinAge INT = 0,
    @MaxAge INT = 100,
    @UserName NVARCHAR(50) = NULL
AS
BEGIN
    SELECT * FROM Users
    WHERE Age BETWEEN @MinAge AND @MaxAge
      AND (@UserName IS NULL OR UserName LIKE '%' + @UserName + '%');
END;

-- 执行存储过程
EXEC usp_SearchUsers @MinAge = 25, @MaxAge = 35;
EXEC usp_SearchUsers @UserName = '张';
```

### 带输出参数的存储过程

```sql
-- 创建带输出参数的存储过程
CREATE PROCEDURE usp_GetUserCount
    @MinAge INT = 0,
    @TotalCount INT OUTPUT
AS
BEGIN
    SELECT @TotalCount = COUNT(*)
    FROM Users
    WHERE Age >= @MinAge;
END;

-- 执行存储过程并获取输出参数
DECLARE @Count INT;
EXEC usp_GetUserCount @MinAge = 25, @TotalCount = @Count OUTPUT;
PRINT '总人数: ' + CAST(@Count AS NVARCHAR(10));
```

## 2. 高级存储过程

### 错误处理

```sql
-- 创建带错误处理的存储过程
CREATE PROCEDURE usp_TransferMoney
    @FromAccountID INT,
    @ToAccountID INT,
    @Amount DECIMAL(10,2)
AS
BEGIN
    SET NOCOUNT ON;
    
    BEGIN TRY
        BEGIN TRANSACTION;
        
        -- 检查余额是否足够
        DECLARE @Balance DECIMAL(10,2);
        SELECT @Balance = Balance FROM Accounts WHERE AccountID = @FromAccountID;
        
        IF @Balance < @Amount
        BEGIN
            RAISERROR('余额不足', 16, 1);
        END
        
        -- 执行转账
        UPDATE Accounts SET Balance = Balance - @Amount WHERE AccountID = @FromAccountID;
        UPDATE Accounts SET Balance = Balance + @Amount WHERE AccountID = @ToAccountID;
        
        COMMIT TRANSACTION;
        
        PRINT '转账成功';
    END TRY
    BEGIN CATCH
        IF @@TRANCOUNT > 0
            ROLLBACK TRANSACTION;
        
        DECLARE @ErrorMessage NVARCHAR(4000) = ERROR_MESSAGE();
        DECLARE @ErrorSeverity INT = ERROR_SEVERITY();
        DECLARE @ErrorState INT = ERROR_STATE();
        
        RAISERROR(@ErrorMessage, @ErrorSeverity, @ErrorState);
    END CATCH
END;
```

### 使用临时表

```sql
-- 使用临时表的存储过程
CREATE PROCEDURE usp_GetUserOrderSummary
AS
BEGIN
    -- 创建临时表
    CREATE TABLE #UserSummary (
        UserID INT,
        UserName NVARCHAR(50),
        TotalOrders INT,
        TotalAmount DECIMAL(10,2)
    );
    
    -- 插入数据到临时表
    INSERT INTO #UserSummary (UserID, UserName, TotalOrders, TotalAmount)
    SELECT 
        u.UserID,
        u.UserName,
        COUNT(o.OrderID) AS TotalOrders,
        ISNULL(SUM(o.Amount), 0) AS TotalAmount
    FROM Users u
    LEFT JOIN Orders o ON u.UserID = o.UserID
    GROUP BY u.UserID, u.UserName;
    
    -- 返回结果
    SELECT * FROM #UserSummary ORDER BY TotalAmount DESC;
    
    -- 清理临时表
    DROP TABLE #UserSummary;
END;
```

### 动态 SQL

```sql
-- 使用动态 SQL 的存储过程
CREATE PROCEDURE usp_SearchDynamic
    @TableName NVARCHAR(100),
    @ColumnName NVARCHAR(100),
    @SearchValue NVARCHAR(100)
AS
BEGIN
    DECLARE @SQL NVARCHAR(MAX);
    
    SET @SQL = N'SELECT * FROM ' + QUOTENAME(@TableName) + 
               N' WHERE ' + QUOTENAME(@ColumnName) + N' LIKE @Value';
    
    EXEC sp_executesql @SQL, N'@Value NVARCHAR(100)', @Value = '%' + @SearchValue + '%';
END;

-- 执行
EXEC usp_SearchDynamic 'Users', 'UserName', '张';
```

## 3. 自定义函数

### 标量函数

```sql
-- 创建标量函数
CREATE FUNCTION dbo.fn_CalculateAge
(
    @BirthDate DATE
)
RETURNS INT
AS
BEGIN
    DECLARE @Age INT;
    SET @Age = DATEDIFF(YEAR, @BirthDate, GETDATE());
    
    -- 精确计算年龄
    IF DATEADD(YEAR, @Age, @BirthDate) > GETDATE()
        SET @Age = @Age - 1;
    
    RETURN @Age;
END;

-- 使用标量函数
SELECT 
    UserID,
    UserName,
    dbo.fn_CalculateAge(BirthDate) AS Age
FROM Users;
```

### 表值函数

```sql
-- 创建内联表值函数
CREATE FUNCTION dbo.fn_GetUsersByAge
(
    @MinAge INT,
    @MaxAge INT
)
RETURNS TABLE
AS
RETURN
(
    SELECT * FROM Users WHERE Age BETWEEN @MinAge AND @MaxAge
);

-- 使用内联表值函数
SELECT * FROM dbo.fn_GetUsersByAge(25, 35);

-- 创建多语句表值函数
CREATE FUNCTION dbo.fn_GetUserOrderDetails
(
    @UserID INT
)
RETURNS @Result TABLE
(
    OrderID INT,
    OrderDate DATETIME,
    Amount DECIMAL(10,2),
    ProductName NVARCHAR(100)
)
AS
BEGIN
    INSERT INTO @Result
    SELECT 
        o.OrderID,
        o.OrderDate,
        o.Amount,
        p.ProductName
    FROM Orders o
    JOIN OrderDetails od ON o.OrderID = od.OrderID
    JOIN Products p ON od.ProductID = p.ProductID
    WHERE o.UserID = @UserID;
    
    RETURN;
END;

-- 使用多语句表值函数
SELECT * FROM dbo.fn_GetUserOrderDetails(1);
```

## 4. 存储过程与函数对比

| 特性 | 存储过程 | 函数 |
|------|----------|------|
| 返回值 | 可以返回多个结果集 | 只能返回一个值或表 |
| 参数类型 | 输入、输出、返回参数 | 只有输入参数 |
| 执行方式 | EXEC 或 CALL | 在 SQL 语句中使用 |
| 修改数据 | 可以 | 不可以 |
| 事务 | 支持 | 不支持 |
| 错误处理 | 支持 | 有限支持 |
| 性能 | 更优 | 一般 |

## 5. Go 调用存储过程示例

```go
package main

import (
    "database/sql"
    "fmt"
    "log"
    
    _ "github.com/microsoft/go-mssqldb"
)

func main() {
    connString := "server=localhost;user id=sa;password=YourPassword;database=TestDB"
    db, err := sql.Open("sqlserver", connString)
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()
    
    // 调用无参数存储过程
    rows, err := db.Query("EXEC usp_GetAllUsers")
    if err != nil {
        log.Fatal(err)
    }
    defer rows.Close()
    
    for rows.Next() {
        var userID int
        var userName string
        var age int
        err := rows.Scan(&userID, &userName, &age)
        if err != nil {
            log.Fatal(err)
        }
        fmt.Printf("%d: %s (%d岁)
", userID, userName, age)
    }
    
    // 调用带参数的存储过程
    var totalAmount float64
    err = db.QueryRow("EXEC usp_GetUserTotalAmount @UserID = @p1", 1).Scan(&totalAmount)
    if err != nil {
        log.Fatal(err)
    }
    fmt.Printf("总金额: %.2f
", totalAmount)
    
    // 调用带输出参数的存储过程
    var count int
    _, err = db.Exec("DECLARE @Count INT; EXEC usp_GetUserCount @MinAge = 25, @TotalCount = @Count OUTPUT; SELECT @Count AS Result")
    if err != nil {
        log.Fatal(err)
    }
}
```

## 6. 管理存储过程和函数

```sql

SELECT 
    name,
    create_date,
    modify_date
FROM sys.objects

SELECT 
    o.name AS ProcedureName,
    m.definition AS Definition
FROM sys.objects o
JOIN sys.sql_modules m ON o.object_id = m.object_id
WHERE o.type = 'P';
```

## 7. 性能优化

### 存储过程优化建议

```sql
-- 1. 使用 SET NOCOUNT ON
CREATE PROCEDURE usp_OptimizedProcedure
AS
BEGIN
    SET NOCOUNT ON;  -- 减少网络流量
    
    -- 你的代码
    SELECT * FROM Users;
END;

-- 2. 使用参数嗅探优化
CREATE PROCEDURE usp_OptimizedSearch
    @UserName NVARCHAR(50)
AS
BEGIN
    SET NOCOUNT ON;
    
    -- 使用局部变量避免参数嗅探问题
    DECLARE @LocalUserName NVARCHAR(50) = @UserName;
    
    SELECT * FROM Users WHERE UserName LIKE '%' + @LocalUserName + '%';
END;

-- 3. 使用 WITH RECOMPILE 处理参数嗅探
CREATE PROCEDURE usp_SearchWithRecompile
    @MinAge INT
WITH RECOMPILE
AS
BEGIN
    SELECT * FROM Users WHERE Age >= @MinAge;
END;
```

## 9. 安全性

### 权限管理

```sql
-- 授予执行权限
GRANT EXECUTE ON usp_GetAllUsers TO Public;

-- 授予特定用户执行权限

REVOKE EXECUTE ON usp_DeleteUser FROM Guest;
```

## 10. 最佳实践

1. **命名规范**：
   - 存储过程：usp_ 前缀（如：usp_GetUsers）
   - 函数：fn_ 前缀（如：fn_CalculateAge）

2. **参数设计**：
   - 提供合理的默认值
   - 避免过多参数（建议不超过 10 个）
   - 使用有意义的参数名

3. **错误处理**：
   - 使用 TRY-CATCH 块
   - 记录错误信息
   - 使用有意义错误消息

4. **性能优化**：
   - 使用 SET NOCOUNT ON
   - 避免在循环中执行 SQL
   - 使用适当的索引
   - 避免使用 SELECT *

5. **文档化**：
   - 添加注释说明功能
   - 记录参数用途
   - 记录修改历史
