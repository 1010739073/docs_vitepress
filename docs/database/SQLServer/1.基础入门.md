# SQL Server 基础入门

## 1. SQL Server 简介

SQL Server 是微软开发的关系型数据库管理系统，广泛应用于企业级应用中。它具有以下特点：

- **安全性高**：完善的安全机制和权限控制
- **性能优异**：高效的查询优化器和索引技术
- **扩展性强**：支持分布式事务和集群部署
- **易用性好**：图形化管理工具和丰富的开发接口

### SQL Server 版本对比

| 版本 | 发布时间 | 主要特性 | 适用场景 |
|------|----------|----------|----------|
| SQL Server 2019 | 2019 | 大数据集群、Python/R集成 | 大型企业 |
| SQL Server 2022 | 2022 | Azure集成、查询优化 | 现代化应用 |
| SQL Server Express | 免费 | 基础功能、4GB限制 | 小型应用 |

## 2. 安装与配置

### Windows 安装

1. 下载 SQL Server 安装包
2. 运行安装程序，选择基本安装
3. 设置 sa 账户密码
4. 安装完成后重启系统

### Linux 安装（以 Ubuntu 为例）

```bash
# 导入公钥
wget -qO- https://packages.microsoft.com/keys/microsoft.asc | sudo apt-key add -

# 添加仓库
sudo add-apt-repository "$(wget -qO- https://packages.microsoft.com/config/ubuntu/20.04/mssql-server-2022.list)"

# 安装 SQL Server
sudo apt-get update
sudo apt-get install -y mssql-server

# 配置 SQL Server
sudo /opt/mssql/bin/mssql-conf setup

# 启动服务
sudo systemctl start mssql-server
sudo systemctl enable mssql-server
```

### 连接数据库

**使用 sqlcmd 工具：**

```bash
sqlcmd -S localhost -U sa -P 'YourPassword'
```

**使用 Go 语言连接：**

```go
package main

import (
    "database/sql"
    "fmt"
    "log"
    
    _ "github.com/microsoft/go-mssqldb"
)

func main() {
    // 连接字符串
    connString := "server=localhost;user id=sa;password=YourPassword;database=master"
    
    // 创建连接
    db, err := sql.Open("sqlserver", connString)
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()
    
    // 测试连接
    err = db.Ping()
    if err != nil {
        log.Fatal(err)
    }
    
    fmt.Println("连接成功!")
}
```

## 3. 数据库和表的基本操作

### 创建数据库

```sql
-- 创建数据库
CREATE DATABASE TestDB;

-- 使用数据库
USE TestDB;

-- 查看所有数据库
SELECT name FROM sys.databases;
```

### 创建表

```sql
-- 创建用户表
CREATE TABLE Users (
    UserID INT PRIMARY KEY IDENTITY(1,1),
    UserName NVARCHAR(50) NOT NULL,
    Email NVARCHAR(100) NOT NULL,
    Age INT,
    CreatedDate DATETIME DEFAULT GETDATE()
);

-- 创建订单表（带外键）
CREATE TABLE Orders (
    OrderID INT PRIMARY KEY IDENTITY(1,1),
    UserID INT NOT NULL,
    OrderDate DATETIME DEFAULT GETDATE(),
    Amount DECIMAL(10, 2),
    FOREIGN KEY (UserID) REFERENCES Users(UserID)
);
```

### 修改表结构

```sql
-- 添加列
ALTER TABLE Users ADD Phone NVARCHAR(20);

-- 修改列
ALTER TABLE Users ALTER COLUMN UserName NVARCHAR(100);

-- 删除列
ALTER TABLE Users DROP COLUMN Phone;

-- 删除表
DROP TABLE IF EXISTS Orders;
```

## 4. CRUD 操作

### 插入数据

```sql
-- 插入单条数据
INSERT INTO Users (UserName, Email, Age)
VALUES ('张三', 'zhangsan@example.com', 25);

-- 插入多条数据
INSERT INTO Users (UserName, Email, Age)
VALUES 
    ('李四', 'lisi@example.com', 30),
    ('王五', 'wangwu@example.com', 28);

-- 插入数据并返回插入的ID
INSERT INTO Users (UserName, Email, Age)
OUTPUT INSERTED.UserID
VALUES ('赵六', 'zhaoliu@example.com', 35);
```

### 查询数据

```sql
-- 基础查询
SELECT * FROM Users;

-- 条件查询
SELECT * FROM Users WHERE Age > 25;

-- 排序查询
SELECT * FROM Users ORDER BY Age DESC;

-- 分页查询
SELECT * FROM Users
ORDER BY UserID
OFFSET 0 ROWS FETCH NEXT 10 ROWS ONLY;

-- 聚合查询
SELECT COUNT(*) AS TotalUsers,
       AVG(Age) AS AvgAge,
       MAX(Age) AS MaxAge,
       MIN(Age) AS MinAge
FROM Users;

-- 分组查询
SELECT Age, COUNT(*) AS Count
FROM Users
GROUP BY Age
HAVING COUNT(*) > 1;
```

### 更新数据

```sql
-- 更新单条数据
UPDATE Users SET Age = 26 WHERE UserID = 1;

-- 更新多条数据
UPDATE Users SET Email = 'new_' + Email WHERE Age > 30;

-- 使用子查询更新
UPDATE Users SET Age = Age + 1 WHERE UserID IN (
    SELECT DISTINCT UserID FROM Orders WHERE Amount > 1000
);
```

### 删除数据

```sql
-- 删除单条数据
DELETE FROM Users WHERE UserID = 1;

-- 删除多条数据
DELETE FROM Users WHERE Age < 20;

-- 使用子查询删除
DELETE FROM Orders WHERE UserID IN (
    SELECT UserID FROM Users WHERE Age > 40
);

-- 清空表（保留表结构）
TRUNCATE TABLE Users;
```

## 5. Go 语言 CRUD 示例

```go
package main

import (
    "database/sql"
    "fmt"
    "log"
    "time"
    
    _ "github.com/microsoft/go-mssqldb"
)

type User struct {
    UserID      int
    UserName    string
    Email       string
    Age         int
    CreatedDate time.Time
}

func main() {
    connString := "server=localhost;user id=sa;password=YourPassword;database=TestDB"
    db, err := sql.Open("sqlserver", connString)
    if err != nil {
        log.Fatal(err)
    }
    defer db.Close()
    
    // 插入数据
    insertUser(db, "测试用户", "test@example.com", 25)
    
    // 查询数据
    users, err := getUsers(db)
    if err != nil {
        log.Fatal(err)
    }
    
    for _, user := range users {
        fmt.Printf("%d: %s (%d岁) %s
", user.UserID, user.UserName, user.Age, user.Email)
    }
    
    // 更新数据
    updateUserAge(db, 1, 26)
    
    // 删除数据
    deleteUser(db, 2)
}

func insertUser(db *sql.DB, userName, email string, age int) {
    query := `INSERT INTO Users (UserName, Email, Age) VALUES (@UserName, @Email, @Age)`
    _, err := db.Exec(query,
        sql.Named("UserName", userName),
        sql.Named("Email", email),
        sql.Named("Age", age))
    if err != nil {
        log.Printf("插入失败: %v", err)
    }
}

func getUsers(db *sql.DB) ([]User, error) {
    query := `SELECT UserID, UserName, Email, Age, CreatedDate FROM Users`
    rows, err := db.Query(query)
    if err != nil {
        return nil, err
    }
    defer rows.Close()
    
    var users []User
    for rows.Next() {
        var user User
        err := rows.Scan(&user.UserID, &user.UserName, &user.Email, &user.Age, &user.CreatedDate)
        if err != nil {
            return nil, err
        }
        users = append(users, user)
    }
    return users, nil
}

func updateUserAge(db *sql.DB, userID, newAge int) {
    query := `UPDATE Users SET Age = @Age WHERE UserID = @UserID`
    _, err := db.Exec(query, sql.Named("Age", newAge), sql.Named("UserID", userID))
    if err != nil {
        log.Printf("更新失败: %v", err)
    }
}

func deleteUser(db *sql.DB, userID int) {
    query := `DELETE FROM Users WHERE UserID = @UserID`
    _, err := db.Exec(query, sql.Named("UserID", userID))
    if err != nil {
        log.Printf("删除失败: %v", err)
    }
}
```

## 6. 数据类型

SQL Server 支持以下主要数据类型：

### 数值类型

| 类型 | 描述 | 范围 |
|------|------|------|
| INT | 整数 | -2^31 到 2^31-1 |
| BIGINT | 大整数 | -2^63 到 2^63-1 |
| SMALLINT | 小整数 | -2^15 到 2^15-1 |
| TINYINT | 微整数 | 0 到 255 |
| DECIMAL(p,s) | 精确数值 | p=总位数，s=小数位数 |
| FLOAT | 浮点数 | 依赖实现 |

### 字符串类型

| 类型 | 描述 | 最大长度 |
|------|------|----------|
| CHAR(n) | 固定长度字符 | 8000 |
| VARCHAR(n) | 可变长度字符 | 8000 |
| NCHAR(n) | Unicode固定长度 | 4000 |
| NVARCHAR(n) | Unicode可变长度 | 4000 |
| TEXT | 大文本 | 2^31-1 |

### 日期时间类型

| 类型 | 描述 | 格式 |
|------|------|------|
| DATE | 日期 | YYYY-MM-DD |
| TIME | 时间 | HH:MM:SS |
| DATETIME | 日期时间 | YYYY-MM-DD HH:MM:SS |
| DATETIME2 | 精确日期时间 | 精确到100纳秒 |
| TIMESTAMP | 时间戳 | 二进制 |

## 7. 常用函数

### 字符串函数

```sql
-- 字符串长度
SELECT LEN('Hello World') AS Length;  -- 返回 11

-- 大小写转换
SELECT UPPER('hello') AS Upper;        -- 返回 HELLO
SELECT LOWER('HELLO') AS Lower;        -- 返回 hello

-- 字符串拼接
SELECT CONCAT('Hello', ' ', 'World') AS Result;  -- 返回 Hello World

-- 字符串截取
SELECT SUBSTRING('Hello World', 1, 5) AS Result;  -- 返回 Hello

-- 去除空格
SELECT LTRIM('  Hello') AS Result;    -- 返回 Hello
SELECT RTRIM('Hello  ') AS Result;    -- 返回 Hello
```

### 数值函数

```sql
-- 四舍五入
SELECT ROUND(3.14159, 2) AS Result;  -- 返回 3.14

-- 向上取整
SELECT CEILING(3.1) AS Result;      -- 返回 4

-- 向下取整
SELECT FLOOR(3.9) AS Result;        -- 返回 3

-- 绝对值
SELECT ABS(-10) AS Result;          -- 返回 10
```

### 日期时间函数

```sql
-- 当前日期时间
SELECT GETDATE() AS CurrentDateTime;
SELECT SYSDATETIME() AS PreciseDateTime;

-- 日期部分
SELECT YEAR(GETDATE()) AS Year;
SELECT MONTH(GETDATE()) AS Month;
SELECT DAY(GETDATE()) AS Day;

-- 日期运算
SELECT DATEADD(DAY, 7, GETDATE()) AS NextWeek;
SELECT DATEDIFF(DAY, '2024-01-01', GETDATE()) AS DaysPassed;
```

## 8. 注意事项

1. **大小写敏感性**：默认情况下，SQL Server 不区分大小写，但可以使用 COLLATE 设置区分大小写
2. **空值处理**：使用 IS NULL 或 IS NOT NULL 检查空值，不能使用 = NULL
3. **字符串引号**：字符串使用单引号，双引号用于标识符
4. **事务隔离级别**：合理设置事务隔离级别，避免死锁
5. **参数化查询**：始终使用参数化查询防止 SQL 注入

## 9. 实践建议

1. **命名规范**：
   - 表名：PascalCase（如：Users）
   - 列名：PascalCase（如：UserName）
   - 存储过程：usp_前缀（如：usp_GetUsers）

2. **性能优化**：
   - 为经常查询的列创建索引
   - 避免使用 SELECT *
   - 合理使用事务，保持事务简短
   - 使用执行计划分析查询性能

3. **安全性**：
   - 使用最小权限原则
   - 定期备份数据
   - 启用审计日志
   - 及时更新补丁
