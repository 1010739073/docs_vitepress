**常见原因及解决方案：**

### 1.2 连接池问题

```sql
-- 查看当前连接
SELECT 
    DB_NAME(dbid) AS DatabaseName,
    COUNT(dbid) AS ConnectionCount,
    loginame AS LoginName
FROM sys.sysprocesses
WHERE dbid > 0
GROUP BY dbid, loginame
ORDER BY ConnectionCount DESC;

    session_id,
    login_time,
    program_name,

    status,
    last_request_start_time,
    last_request_end_time
FROM sys.dm_exec_sessions s
JOIN sys.dm_exec_connections c ON s.session_id = c.session_id
WHERE s.session_id > 50;

-- 查看睡眠连接（连接池未释放）
SELECT 
    session_id,
    login_time,
    last_request_start_time,
    DATEDIFF(MINUTE, last_request_start_time, GETDATE()) AS IdleMinutes,
    program_name,
    login_name
FROM sys.dm_exec_sessions
WHERE status = 'sleeping'
AND DATEDIFF(MINUTE, last_request_start_time, GETDATE()) > 30;
```

## 二、性能问题

### 2.1 CPU 使用率过高

```sql
-- 查看当前CPU使用率
SELECT 
    SQLProcessUtilization AS SQLCPU,
    SystemIdle AS Idle,
    100 - SystemIdle - SQLProcessUtilization AS OtherProcessCPU
FROM (
    SELECT 
        TOP 1 
        record_id,
        [SQLProcessUtilization],
        [SystemIdle]
    FROM (
        SELECT 
            record.value('(./Record/@id)[1]', 'int') AS record_id,
            record.value('(./Record/SchedulerMonitorEvent/SystemHealth/SystemIdle)[1]', 'int') AS SystemIdle,
            record.value('(./Record/SchedulerMonitorEvent/SystemHealth/ProcessUtilization)[1]', 'int') AS SQLProcessUtilization
        FROM (
            SELECT TOP 30 CONVERT(XML, record) AS record
            FROM sys.dm_os_ring_buffers
            WHERE ring_buffer_type = N'RING_BUFFER_SCHEDULER_MONITOR'
            AND record LIKE '%<SystemIdle>%'
            ORDER BY timestamp DESC
        ) AS x
    ) AS y
    ORDER BY record_id DESC
) AS z;

-- 查看高CPU消耗的查询
SELECT TOP 20
    qs.execution_count,
    qs.total_worker_time / 1000 AS total_cpu_time_ms,
    qs.total_elapsed_time / 1000 AS total_duration_ms,
    qs.total_logical_reads,
    qs.total_logical_writes,
    SUBSTRING(qt.text, (qs.statement_start_offset/2)+1,
        ((CASE qs.statement_end_offset
            WHEN -1 THEN DATALENGTH(qt.text)
            ELSE qs.statement_end_offset
        END - qs.statement_start_offset)/2)+1) AS query_text,
    qp.query_plan
FROM sys.dm_exec_query_stats qs
CROSS APPLY sys.dm_exec_sql_text(qs.sql_handle) qt
CROSS APPLY sys.dm_exec_query_plan(qs.plan_handle) qp
ORDER BY qs.total_worker_time DESC;
```

### 2.4 阻塞与死锁

```sql
-- 查看当前阻塞
SELECT 
    blocking_session_id,
    session_id,
    wait_type,
    wait_time,
    wait_resource,
    DB_NAME(database_id) AS DatabaseName
FROM sys.dm_exec_requests
WHERE blocking_session_id <> 0;

-- 查看阻塞链
WITH BlockingCTE AS (
    SELECT 
        blocking_session_id,
        session_id,
        wait_type,
        wait_time,
        wait_resource,
        0 AS Level
    FROM sys.dm_exec_requests
    WHERE blocking_session_id = 0 AND session_id IN (
        SELECT blocking_session_id FROM sys.dm_exec_requests WHERE blocking_session_id <> 0
    )
    
    UNION ALL
    
    SELECT 
        r.blocking_session_id,
        r.session_id,
        r.wait_type,
        r.wait_time,
        r.wait_resource,
        b.Level + 1
    FROM sys.dm_exec_requests r
    INNER JOIN BlockingCTE b ON r.blocking_session_id = b.session_id
)
SELECT * FROM BlockingCTE ORDER BY Level, session_id;

-- 查看死锁历史
SELECT 
    xed.value('@timestamp', 'datetime') AS DeadlockTime,
    xed.query('.') AS DeadlockGraph
FROM (
    SELECT CAST(target_data AS XML) AS TargetData
    FROM sys.dm_xe_session_targets st
    JOIN sys.dm_xe_sessions s ON s.address = st.event_session_address
    WHERE name = 'system_health'
) AS Data
CROSS APPLY TargetData.nodes('/RingBufferTarget/event[@name="xml_deadlock_report"]') AS XEventData(xed);
```

## 四、查询性能问题

### 4.1 参数嗅探问题

```sql
-- 问题现象：同一存储过程不同参数执行时间差异大

-- 解决方案1：使用本地变量
CREATE PROCEDURE GetOrdersByStatus @Status VARCHAR(20)
AS
BEGIN
    DECLARE @LocalStatus VARCHAR(20) = @Status;
    SELECT * FROM Orders WHERE Status = @LocalStatus;
END;

-- 解决方案2：使用 OPTION(RECOMPILE)
CREATE PROCEDURE GetOrdersByStatus @Status VARCHAR(20)
AS
BEGIN
    SELECT * FROM Orders WHERE Status = @Status
    OPTION (RECOMPILE);
END;

-- 解决方案3：使用 OPTIMIZE FOR UNKNOWN
CREATE PROCEDURE GetOrdersByStatus @Status VARCHAR(20)
AS
BEGIN
    SELECT * FROM Orders WHERE Status = @Status
    OPTION (OPTIMIZE FOR (@Status UNKNOWN));
END;

-- 解决方案4：使用提示
CREATE PROCEDURE GetOrdersByStatus @Status VARCHAR(20)
AS
BEGIN
    SELECT * FROM Orders WITH (INDEX(IX_Orders_Status))
    WHERE Status = @Status;
END;
```

### 6.2 SQL 注入防护

```sql
-- 不安全的写法（字符串拼接）
CREATE PROCEDURE UnsafeSearch @UserName NVARCHAR(50)
AS
BEGIN
    DECLARE @SQL NVARCHAR(MAX);
    SET @SQL = 'SELECT * FROM Users WHERE UserName = ''' + @UserName + '''';
    EXEC sp_executesql @SQL;  -- 危险！
END;

-- 安全的写法（参数化查询）
CREATE PROCEDURE SafeSearch @UserName NVARCHAR(50)
AS
BEGIN
    SELECT * FROM Users WHERE UserName = @UserName;
END;

-- 动态SQL的安全写法
CREATE PROCEDURE SafeDynamicSearch @ColumnName NVARCHAR(50), @Value NVARCHAR(100)
AS
BEGIN
    DECLARE @SQL NVARCHAR(MAX);
    -- 使用QUOTENAME防止注入
    SET @SQL = N'SELECT * FROM Users WHERE ' + QUOTENAME(@ColumnName) + ' = @ValueParam';
    EXEC sp_executesql @SQL, N'@ValueParam NVARCHAR(100)', @ValueParam = @Value;
END;
```

### 7.2 监控脚本

```sql
-- 创建监控存储过程
CREATE PROCEDURE sp_DatabaseHealthCheck
AS
BEGIN
    SET NOCOUNT ON;
    
    CREATE TABLE #HealthCheck (
        CheckItem NVARCHAR(100),
        Status NVARCHAR(20),
        Details NVARCHAR(MAX)
    );
    
    -- 检查数据库状态
    INSERT INTO #HealthCheck
    SELECT 
        'Database Status',
        CASE WHEN state_desc = 'ONLINE' THEN 'OK' ELSE 'WARNING' END,
        name + ': ' + state_desc
    FROM sys.databases
    WHERE state_desc <> 'ONLINE';
    
    -- 检查磁盘空间
    INSERT INTO #HealthCheck
    SELECT 
        'Disk Space',
        CASE WHEN free_space_mb > 10240 THEN 'OK' ELSE 'WARNING' END,
        drive + ': ' + CAST(free_space_mb / 1024 AS VARCHAR) + ' GB free'
    FROM (
        SELECT DISTINCT 
            LEFT(physical_name, 1) AS drive,
            (CAST(SUM(free_space) AS BIGINT) / 1024 / 1024) AS free_space_mb
        FROM sys.master_files mf
        CROSS APPLY sys.dm_os_volume_stats(mf.database_id, mf.file_id)
        GROUP BY LEFT(physical_name, 1)
    ) t
    WHERE free_space_mb < 10240;
    
    -- 检查阻塞
    INSERT INTO #HealthCheck
    SELECT 
        'Blocking',
        'WARNING',
        'Session ' + CAST(blocking_session_id AS VARCHAR) + ' blocking ' + CAST(session_id AS VARCHAR)
    FROM sys.dm_exec_requests
    WHERE blocking_session_id <> 0;
    
    -- 检查结果
    SELECT * FROM #HealthCheck;
    
    DROP TABLE #HealthCheck;
END;
```
